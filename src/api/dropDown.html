Example
items 参数的配置:
[
    {"key": "fish", value: "鱼", html: "<b>一只鱼</b>"},
    {"key": "cat", value: "猫", html: "<b>一条猫</b>"}
]
option.fun 的使用：
// items 的参数
[
    {"name": "wang", "sex": 0},
    {"name": "li", "sex": 1}
]
// option.fun
option.fun = function(index, obj) {
   // obj 为 items 中的一条数据
   var html = obj.name + (obj.sex ? "is man" : "is woman");
   
   // 返回值为 DropItem 数据格式
   return {
       key: index,  // 对数据的唯一标识
       value: obj.name, // 记录原始数据的值
       html: html // 用于下拉控件显示时的 html
   }
}
